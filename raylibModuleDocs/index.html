class Window(int width, int height, const char *title); // initializes the window.

void setIcon(const char *path); // changes the window icon.

void TargetFPS(int fps); // sets a max fps.

bool isClosing(); // tells you if the window is closing or not.

class Renderer;

void InitRenderer(); // starts the renderer, put it in loop.

void disableRenderer(); // ends the renderer, put it after InitRenderer().

void clearBG(Color color); // clears the background with a color.

void renderFPS(int x, int y); // displays the game fps.

class Sprite(float xPos, float yPos, float width,float height); // initializes the sprite.

void InitTexture(const char *path); // creates a texture for sprite.

void render(Color color); // renders the sprite with specified color.

void unload(); // unloads the initialized texture from memory. put it after the loop.

void updateTexture(); //updates the texture size to fit the collider.

float getX(); // gets the x value of the sprite

float getY(); // gets the y value of the sprite

float setX(float x); // sets the x value of the sprite

float setY(float y); // sets the y value of the sprite

class Event; // initializes the Event Manager.

bool KeyDown(int key); // detects if key is down.

bool KeyPressed(int key); // detects if key is pressed.

bool KeyReleased(int key); // detects if key is released.

bool KeyUp(int key); // detects if key is up.
